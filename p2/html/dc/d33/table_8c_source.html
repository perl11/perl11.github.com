362 PN potion_tuple_bsearch(Potion *P, PN cl, PN self, PN x) {
363  struct PNTuple *t = PN_GET_TUPLE(self);
364  PNUniq xu = PN_UNIQ(x);
365  long i = 0, j = t->len - 1;
366  while (i <= j) {
367  long m = j + ((i - j) / 2);
368  PNUniq u = PN_UNIQ(t->set[m]);
369  if (u == xu)
370  return PN_NUM(m);
371  else if (u > xu)
372  j = m - 1;
373  else
374  i = m + 1;
375  }
376  return PN_NUM(-1);
