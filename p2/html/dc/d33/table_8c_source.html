493 static void potion_sort_internal(Potion *P, PN cl, PN self,
494  PN_SIZE from,
495  PN_SIZE to,
496  PN cmp)
497 {
498 #ifdef DEBUG
499  if (PN_TTUPLE != PN_TYPE(self)) potion_fatal("Invalid type");
500 #endif
501  if (from < to) {
502  struct PNTuple *t = PN_GET_TUPLE(self);
503  // which pivot? first is worst case if already sorted.
504  // random, last, middle or best: median-of-3.
505  PN_SIZE i, index = from + (to - from)/2;
506  PN pivot = GET(index);
507  SWAP(index, to);
508  index = from;
509  // partition the portion of the tuple between indexes from and to,
510  // inclusively, by moving all elements less than pivot before
511  // the pivot, and the equal or greater elements after it.
512  if (cmp == PN_NIL) { // default: sort by uniq, not value
513  for (i=from; i < to-1; i++) { // from ≤ i < to
514  if (PN_UNIQ(GET(i)) <= PN_UNIQ(pivot)) { SWAP(i, index); index++; }
515  }
516  } else if (cmp == PN_TRUE) { // sort by ascending number
517  for (i=from; i < to; i++) {
518  if (GET(i) <= pivot) {
519  SWAP(i, index); index++;
520  }
521  }
522  } else if (cmp == PN_FALSE) { // sort by descending number
523  for (i=from; i < to; i++) {
524  if (GET(i) > pivot) { SWAP(i, index); index++;
525  }
526  }
527  } else {
528  vPN(Closure) c = PN_CLOSURE(cmp);
529  for (i=from; i < to; i++) { // call cmp
530  if (PN_INT(c->method(P, cl, cmp, GET(i), pivot)) > 0)
531  { SWAP(i, index); index++; }
532  }
533  }
534  SWAP(index, to); // Move pivot element back to its final place
535 
536  if (index > 0)
537  potion_sort_internal(P,cl,self, from, index-1, cmp);
538  potion_sort_internal(P,cl,self, index+1, to, cmp);
539  PN_TOUCH(self);
540  }
541 }

574 PN potion_tuple_ins_sort(Potion *P, PN cl, PN self, PN cmp) {
575  struct PNTuple *t = PN_GET_TUPLE(self);
576  DBG_CHECK_TYPE(t,PN_TTUPLE);
577  unsigned long i, j;
578  vPN(Closure) c;
579  if (t->len < MAX_INS_SORT) {
580  // simple insertion sort for smaller arrays (<13)
581  if (cmp == PN_NIL) { // default: sort by uniq, not value
582  for (i = 1; i < t->len; i++) {
583  j = i;
584  while (j > 0 && PN_UNIQ(GET(j-1)) > PN_UNIQ(GET(j))) {
585  SWAP(j, j-1);
586  j--;
587  }
588  }
589  }
590  else if (PN_IS_CLOSURE(cmp)) {
591  c = PN_CLOSURE(cmp);
592  for (i = 1; i < t->len; i++) {
593  j = i;
594  while (j > 0 && PN_INT(c->method(P, cl, cmp, GET(j-1), GET(j))) > 0) {
595  SWAP(j, j-1);
596  j--;
597  }
598  }
599  }
600  else if (cmp == PN_TRUE) {
601  for (i = 1; i < t->len; i++) {
602  j = i;
603  while (j > 0 && GET(j-1) > GET(j)) {
604  SWAP(j, j-1);
605  j--;
606  }
607  }
608  }
609  else if (cmp == PN_FALSE) {
610  for (i = 1; i < t->len; i++) {
611  j = i;
612  while (j > 0 && GET(j-1) < GET(j)) {
613  SWAP(j, j-1);
614  j--;
615  }
616  }
617  }
618  else {
619  potion_fatal("sort: invalid cmp type");
620  }
621  PN_TOUCH(self);
622  }
623  else {
624  if (cmp != PN_NIL && !PN_IS_BOOL(cmp) && !PN_IS_CLOSURE(cmp))
625  potion_fatal("sort: invalid cmp type");
626  potion_sort_internal(P, cl, self, 0, t->len-1, cmp);
627  }
628  return self;
629 }

631 static PN potion_tuple_cmp(Potion *P, PN cl, PN self, PN value) {
632  DBG_CHECK_TYPE(self,PN_TTUPLE);
633  switch (potion_type(value)) {
634  case PN_TBOOLEAN: // false < () < true
635  return value == PN_FALSE ? -1 : 1;
636  case PN_TNIL:
637  return -1; //nil < () < (...)
638  case PN_TTUPLE: // recurse
639  if(PN_TUPLE_LEN(self) && PN_TUPLE_LEN(value)) {
640  PN cmp;
641  if ((cmp = potion_send(potion_tuple_first(P,cl,self), PN_cmp,
642  potion_tuple_first(P,cl,value)))
643  == PN_ZERO)
644  {
645  PN t1 = potion_tuple_clone(P,cl,self);
646  PN t2 = potion_tuple_clone(P,cl,value);
647  potion_tuple_pop(P,cl,t1);
648  potion_tuple_pop(P,cl,t2);
649  return potion_send(t1, PN_cmp, t2);
650  }
651  else {
652  return cmp;
653  }
654  }
655  else {
656  if (PN_TUPLE_LEN(value)) return -1;
657  else if (PN_TUPLE_LEN(self)) return 1;
658  else return 0;
659  }
660  default:
661  potion_fatal("Invalid tuple cmp type");
662  return 0;
663  }
664 }
