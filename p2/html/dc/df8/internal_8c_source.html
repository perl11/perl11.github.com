280 void potion_dump_stack(Potion *P) {
281  PN_SIZE n;
282  PN *end, *ebp, *start = P->mem->cstack;
283  struct PNMemory *M = P->mem;
284  POTION_ESP(&end);
285  POTION_EBP(&ebp);
286 #if POTION_STACK_DIR > 0
287  n = end - start;
288 #else
289  n = start - end + 1;
290  start = end;
291  end = M->cstack;
292 #endif
293 
294  printf("-- dumping %u stack from %p to %p --\n", n, start, end);
295  printf(" ebp = %p, *ebp = %lx\n", ebp, *ebp);
296  while (n--) {
297  printf(" stack(%u) = %lx", n, *start);
298  if (IS_GC_PROTECTED(*start))
299  printf(" gc ");
300  else if (IN_BIRTH_REGION(*start))
301  printf(" gc(0) ");
302  else if (IN_OLDER_REGION(*start))
303  printf(" gc(1) ");
304  if (*start == 0)
305  printf(" nil\n");
306  else if (*start & 1)
307  printf(" %ld INT\n", PN_INT(*start));
308  else if (*start & 2)
309  printf(" %s BOOL\n", *start == 2 ? "false" : "true");
310  else
311  printf("\n");
312  start++;
313  }
314 }
