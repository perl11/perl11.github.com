30 PN potion_file_new(Potion *P, PN cl, PN self, PN path, PN modestr) {
31  int fd;
32  mode_t mode;
33  if (strcmp(PN_STR_PTR(modestr), "r") == 0) {
34  mode = O_RDONLY;
35  } else if (strcmp(PN_STR_PTR(modestr), "r+") == 0) {
36  mode = O_RDWR;
37  } else if (strcmp(PN_STR_PTR(modestr), "w") == 0) {
38  mode = O_WRONLY | O_TRUNC | O_CREAT;
39  } else if (strcmp(PN_STR_PTR(modestr), "w+") == 0) {
40  mode = O_RDWR | O_TRUNC | O_CREAT;
41  } else if (strcmp(PN_STR_PTR(modestr), "a") == 0) {
42  mode = O_WRONLY | O_CREAT | O_APPEND;
43  } else if (strcmp(PN_STR_PTR(modestr), "a+") == 0) {
44  mode = O_RDWR | O_CREAT | O_APPEND;
45  } else {
46  // invalid mode
47  return PN_NIL;
48  }
49  if ((fd = open(PN_STR_PTR(path), mode, 0755)) == -1) {
50  perror("open");
51  // TODO: error, perm
52  return PN_NIL;
53  }
54  ((struct PNFile *)self)->fd = fd;
55  ((struct PNFile *)self)->path = path;
56  ((struct PNFile *)self)->mode = mode;
