621 static inline void *potion_gc_alloc(Potion *P, PNType vt, int siz) {
622  struct PNMemory *M = P->mem;
623  struct PNObject *res = 0;
624  if (siz < sizeof(struct PNFwd))

642  struct PNMemory *M = P->mem;

647  return;
648  *(M->birth_storeptr--) = (void *)x;

653 static inline void *potion_gc_realloc(Potion *P, PNType vt, struct PNObject * volatile obj, PN_SIZE sz) {
654  void *dst = 0;
655  PN_SIZE oldsz = 0;
656 
657  if (obj != NULL) {

659  if (oldsz >= sz)
