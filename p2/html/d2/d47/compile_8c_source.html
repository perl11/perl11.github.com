500  PN_PUSH(ctor, PN_AST(EXPR, PN_TUP(PN_AST(MESSAGE, potion_str(P, "self")))));
501  breg++;
502  PN_BLOCK(breg, t->a[2], t->a[1]);
503  }
504  PN_ASM2(OP_CLASS, reg, breg);
505  } else if (t->part == AST_MESSAGE && (t->a[0] == PN_while || t->a[0] == PN_loop)) {
506  int jmp1 = 0, jmp2 = PN_OP_LEN(f->asmb); breg++;
507  struct PNLoop l; l.bjmpc = 0; l.cjmpc = 0;
508  int i;
509  if (t->a[0] == PN_while) {
510  // TODO: error if args to `loop`?
511  PN_ARG_TABLE(t->a[1], breg, 0);
512  jmp1 = PN_OP_LEN(f->asmb);
513  PN_ASM2(OP_NOTJMP, breg, 0);
514  }
515  potion_source_asmb(P, f, &l, 0, (struct PNSource *)t->a[2], reg);
516  PN_ASM1(OP_JMP, (jmp2 - PN_OP_LEN(f->asmb)) - 1);
517  if (t->a[0] == PN_while) {
518  PN_OP_AT(f->asmb, jmp1).b = (PN_OP_LEN(f->asmb) - jmp1) - 1;
519  }
520  for (i = 0; i < l.bjmpc; i++) {
521  PN_OP_AT(f->asmb, l.bjmps[i]).a = (PN_OP_LEN(f->asmb) - l.bjmps[i]) - 1;
522  }
523  for (i = 0; i < l.cjmpc; i++) {
524  PN_OP_AT(f->asmb, l.cjmps[i]).a = (jmp2 - l.cjmps[i]) - 1;
525  }
526  } else if (t->part == AST_MESSAGE && t->a[0] == PN_return) {
527  PN_ARG_TABLE(t->a[1], reg, 0);
528  PN_ASM1(OP_RETURN, reg);
529  } else if (t->part == AST_MESSAGE && t->a[0] == PN_break) {
530  if (loop != NULL) {
531  loop->bjmps[loop->bjmpc++] = PN_OP_LEN(f->asmb);
532  PN_ASM1(OP_JMP, 0);
533  } else {
534  // TODO: Report error: 'break' outside of loop.
535  }
536  } else if (t->part == AST_MESSAGE && t->a[0] == PN_continue) {
537  if (loop != NULL) {
538  loop->cjmps[loop->cjmpc++] = PN_OP_LEN(f->asmb);
539  PN_ASM1(OP_JMP, 0);
540  } else {
541  // TODO: Report error: 'continue' outside of loop.
542  }
543  } else if (t->part == AST_MESSAGE && t->a[0] == PN_self) {
544  PN_ASM1(OP_SELF, reg);
545  } else {
546  u8 opcode = OP_GETUPVAL;
547  PN_SIZE num = PN_NONE;
548  if (count == 0 && t->part == AST_MESSAGE) {
549  num = PN_UPVAL(t->a[0]);
550  if (num == PN_NONE) {
551  num = PN_GET(f->locals, t->a[0]);
552  opcode = OP_GETLOCAL;
553  }
554  }

556  if (num == PN_NONE && t->a[0] != PN_NIL) {
557  u8 oreg = ++breg;
558  int jmp = 0;
559  num = PN_PUT(f->values, t->a[0]);
560  if (count == 0) {
561  PN_ASM1(OP_SELF, oreg);
562  } else {
563  PN_ASM2(OP_MOVE, oreg, reg);
564  }
565  PN_ASM2(OP_LOADK, reg, num);
566  PN_ASM2(t->a[1] != PN_NIL || t->a[2] != PN_NIL ? OP_MESSAGE : OP_BIND, reg, breg);
567  if (t->part == AST_QUERY && t->a[1] != PN_NIL) {
568  jmp = PN_OP_LEN(f->asmb);
569  PN_ASM2(OP_NOTJMP, reg, 0);
570  }
571 #define LOAD_ARG() PN_ARG_TABLE(t->a[1], breg, 1)
572  if (arg) {
573  u8 part1 = ((struct PNSource *)(t->a[1]))->part;
574  if (part1 == AST_VALUE || (part1 == AST_LIST && t->a[2] == PN_NIL)) {
575  LOAD_ARG();
576  }
577  }
578  if (t->a[2] != PN_NIL && ((struct PNSource *)(t->a[2]))->part == AST_PROTO) {
579  vPN(Source) t2 = (struct PNSource *)(t->a[2]);
580  breg++;
581  PN_BLOCK(breg, t2->a[1], t2->a[0]);
582  } else {
583  if (t->a[1] == PN_NIL && t->a[2] == PN_NIL)
584  LOAD_ARG();
585  if (t->a[2] != PN_NIL && ((struct PNSource *)(t->a[2]))->part == AST_BLOCK) {
586  breg++;
587  PN_BLOCK(breg, t->a[2], t->a[1]);
588  }
589  }
590 #undef LOAD_ARG
591  if (t->part == AST_MESSAGE) {
592  PN_ASM2(OP_CALL, reg, breg);
593  } else
594  if (t->a[1] != PN_NIL) {
595  PN_ASM2(OP_CALL, reg, breg);
596  PN_OP_AT(f->asmb, jmp).b = (PN_OP_LEN(f->asmb) - jmp) - 1;
597  } else {
598  PN_ASM2(OP_TEST, reg, breg);
599  }
600  } else {
601  if (num != PN_NONE)
602  PN_ASM2(opcode, reg, num);
603  if (call) {
604  PN_ASM1(OP_SELF, ++breg);
605  PN_ARG_TABLE(t->a[1], breg, 1);
606  if (t->a[2] != PN_NIL) {
607  breg++;
608  PN_BLOCK(breg, t->a[2], PN_NIL);
609  }
610  PN_ASM2(OP_CALL, reg, breg);
611  }
612  }
613  }
614  PN_REG(f, breg);
615  }
616  break;
617 
618  case AST_PATH:
619  case AST_PATHQ: {
620  PN_SIZE num = PN_PUT(f->values, t->a[0]);
621  if (count == 0) {
622  PN_PUT(f->paths, PN_NUM(num));
623  PN_ASM1(OP_SELF, reg);
624  }
625  PN_ASM2(OP_LOADK, reg + 1, num);
626  PN_ASM2(OP_GETPATH, reg, reg + 1);
627  if (t->part == AST_PATHQ)
628  PN_ASM2(OP_TEST, reg, reg);
629  PN_REG(f, reg + 1);
630  }
631  break;
632 
633  case AST_LICK: {
634  u8 breg = reg;
635  PN_SIZE num = PN_PUT(f->values, t->a[0]);
636  PN_ASM2(OP_LOADK, reg, num);
637  if (t->a[1] != PN_NIL)
638  potion_source_asmb(P, f, loop, 0, (struct PNSource *)t->a[1], ++breg);
639  else if (t->a[2] != PN_NIL)
640  PN_ASM2(OP_LOADPN, ++breg, PN_NIL);
641  if (t->a[2] != PN_NIL)
642  potion_source_asmb(P, f, loop, 0, (struct PNSource *)t->a[2], ++breg);
643  PN_ASM2(OP_NEWLICK, reg, breg);
644  PN_REG(f, breg);
645  }
646  break;
647 
648  case AST_LIST:
649  PN_ASM1(OP_NEWTUPLE, reg);
650  if (t->a[0] != PN_NIL) {
651  PN_TUPLE_EACH(t->a[0], i, v, {
652  if (PN_PART(v) == AST_ASSIGN) {
653  vPN(Source) lhs = (struct PNSource *)PN_S(v, 0);
654  if (lhs->part == AST_EXPR && PN_TUPLE_LEN(lhs->a[0]) == 1)
655  {
656  lhs = (struct PNSource *)PN_TUPLE_AT(lhs->a[0], 0);
657  if (lhs->part == AST_MESSAGE) {
658  PN_SIZE num = PN_PUT(f->values, lhs->a[0]);
659  PN_ASM2(OP_LOADK, reg + 1, num);
660  lhs = NULL;
661  }
662  }
663 
664  if (lhs != NULL)
665  potion_source_asmb(P, f, loop, 0, (struct PNSource *)lhs, reg + 1);
666 
667  potion_source_asmb(P, f, loop, 0, (struct PNSource *)PN_S(v, 1), reg + 2);
668  PN_ASM2(OP_SETTABLE, reg, reg + 2);
669  PN_REG(f, reg + 2);
670  } else {
671  potion_source_asmb(P, f, loop, 0, (struct PNSource *)v, reg + 1);
672  PN_ASM2(OP_SETTUPLE, reg, reg + 1);
673  PN_REG(f, reg + 1);
674  }
675  });
676  }
677  break;
678  }

691  PN_TUPLE_EACH(t->a[0], i, v, {

710  }
711  });
