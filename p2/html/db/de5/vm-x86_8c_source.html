200 void potion_x86_stack(Potion *P, struct PNProto * volatile f, PNAsm * volatile *asmp, long need) {
201  /* maintain 16-byte stack alignment. OS X in particular requires it, because
202  * it expects to be able to use movdqa on things on the stack.
203  * we factor in the offset from our saved ebp and return address, so that
204  * adds 8 for x86 and 0 (mod 16) for x86_64. */
205  int rsp = X86C(16,0)+((need-X86C(8,0)+15)&~(15));
206  if (rsp >= 0x80) {
207  X86_PRE(); ASM(0x81); ASM(0xEC); ASMI(rsp); /* sub rsp, %esp */
208  } else {
209  X86_PRE(); ASM(0x83); ASM(0xEC); ASM(rsp); /* sub rsp, %esp */
210  }
