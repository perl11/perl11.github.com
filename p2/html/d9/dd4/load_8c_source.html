18 PN potion_load_code(Potion *P, const char *filename) {
19  PN buf, code;
20  struct stat stats;
21  int fd = -1;
22  PN result = PN_NIL;
23  if (stat(filename, &stats) == -1) {
24  fprintf(stderr, "** %s does not exist.", filename);
25  return PN_NIL;
26  }
27  fd = open(filename, O_RDONLY | O_BINARY);
28  if (fd == -1) {
29  fprintf(stderr, "** could not open %s. check permissions.", filename);
30  return PN_NIL;
31  }
32  buf = potion_bytes(P, stats.st_size);
33  if (read(fd, PN_STR_PTR(buf), stats.st_size) == stats.st_size) {
34  PN_STR_PTR(buf)[stats.st_size] = '\0';
35  code = potion_source_load(P, PN_NIL, buf);
36  if (!PN_IS_PROTO(code)) {
37  result = potion_run(P, potion_send(
38  potion_parse(P, buf, (char *)filename),
39  PN_compile, potion_str(P, filename), PN_NIL),
40  POTION_JIT);
41  }
42  } else {
43  fprintf(stderr, "** could not read entire file: %s.", filename);
44  }
45  close(fd);
46  return result;
47 }

67  void *handle = dlopen(filename, RTLD_LAZY);

69  char *err, *init_func_name;

120 char *potion_find_file(char *str, PN_SIZE str_len) {
121  char *r = NULL;
122  struct stat st;
123  if (!str_len) str_len = strlen(str);
124  PN_TUPLE_EACH(pn_loader_path, i, prefix, {
125  PN_SIZE prefix_len = PN_STR_LEN(prefix);
126  char dirname[prefix_len + 1 + str_len + 1];
127  char *str_pos = dirname + prefix_len + 1;
128  char *dot;
129  const char *ext;
130  memcpy(str_pos, str, str_len);
131  dot = memchr(str, '.', str_len);
132  if (dot == NULL)
133  dirname[sizeof(dirname) - 1] = '\0';
134  else
135  *dot = '\0';
136  memcpy(dirname, PN_STR_PTR(prefix), prefix_len);
137  dirname[prefix_len] = '/';
138  if (stat(dirname, &st) == 0 && S_ISREG(st.st_mode)) {
139  if (asprintf(&r, "%s", dirname) == -1) potion_allocation_error();
140  break;
141  } else if ((ext = find_extension(dirname)) != NULL) {
142  if (asprintf(&r, "%s%s", dirname, ext) == -1) potion_allocation_error();
143  break;
144  } else {
145  char *file;
146  if ((file = strrchr(str, '/')) == NULL)
147  file = str;
148  else
149  file++;
150  if (asprintf(&r, "%s/%s", dirname, file) == -1) potion_allocation_error();
151  if (stat(r, &st) != 0 || !S_ISREG(st.st_mode)) {
152  int r_len = prefix_len + 1 + str_len * 2 + 1;
153  if ((ext = find_extension(r)) == NULL) { free(r); r = NULL; continue; }
154  r = realloc(r, r_len + strlen(ext));
155  if (r == NULL) potion_allocation_error();
156  strcpy(r + r_len, ext);
157  }
158  break;
159  }
160  });
161  return r;
162 }

166  char *filename = potion_find_file(PN_STR_PTR(file), PN_STR_LEN(file)), *file_ext;
167  PN result = PN_NIL;

173  while (*--file_ext != '.' && file_ext >= filename);
174  if (file_ext++ != filename) {
175  if (strcmp(file_ext, "pn") == 0)

181  else

183  } else {
184  fprintf(stderr, "** no file extension: %s\n", filename);
185  }
