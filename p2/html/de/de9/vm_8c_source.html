114  if (PN_IS_STR(v)) p2args++;

162  CASE_OP(LOADPN, (P, f, &asmb, pos)) // load a value into a register
163  CASE_OP(LOADK, (P, f, &asmb, pos, need)) // load a constant into a register
164  CASE_OP(SELF, (P, f, &asmb, pos, need)) // prepare an object method for calling
165  // R(A+1) := R(B); R(A) := R(B)[RK(C)]

167  CASE_OP(SETLOCAL, (P, f, &asmb, pos, regs))// write a register value into a local
168  CASE_OP(GETUPVAL, (P, f, &asmb, pos, lregs))// read an upvalue (upper scope)
169  CASE_OP(SETUPVAL, (P, f, &asmb, pos, lregs))// write to an upvalue
170  CASE_OP(GLOBAL, (P, f, &asmb, pos, need)) // returns a global (for get or set)
171  CASE_OP(NEWTUPLE, (P, f, &asmb, pos, need))// create tuple
172  CASE_OP(SETTUPLE, (P, f, &asmb, pos, need))// write register into tuple key
173  CASE_OP(SETTABLE, (P, f, &asmb, pos, need))// write register into a table entry
174  CASE_OP(NEWLICK, (P, f, &asmb, pos, need))// create lick. R(A) := {} (size = B,C)
175  CASE_OP(GETPATH, (P, f, &asmb, pos, need))// read obj field into register
176  CASE_OP(SETPATH, (P, f, &asmb, pos, need))// write into obj field
177  CASE_OP(ADD, (P, f, &asmb, pos, need)) // a = b + c
178  CASE_OP(SUB, (P, f, &asmb, pos, need)) // a = b - c
179  CASE_OP(MULT, (P, f, &asmb, pos, need))
180  CASE_OP(DIV, (P, f, &asmb, pos, need))
181  CASE_OP(REM, (P, f, &asmb, pos, need))
182  CASE_OP(POW, (P, f, &asmb, pos, need))
183  CASE_OP(NEQ, (P, f, &asmb, pos))
184  CASE_OP(EQ, (P, f, &asmb, pos)) // if ((RK(B) == RK(C)) ~= A) then PC++
185  CASE_OP(LT, (P, f, &asmb, pos)) // if ((RK(B) < RK(C)) ~= A) then PC++
186  CASE_OP(LTE, (P, f, &asmb, pos)) // if ((RK(B) <= RK(C)) ~= A) then PC++
187  CASE_OP(GT, (P, f, &asmb, pos))
188  CASE_OP(GTE, (P, f, &asmb, pos))
189  CASE_OP(BITN, (P, f, &asmb, pos, need))
190  CASE_OP(BITL, (P, f, &asmb, pos, need))
191  CASE_OP(BITR, (P, f, &asmb, pos, need))
192  CASE_OP(DEF, (P, f, &asmb, pos, need)) // define a method for an object
193  CASE_OP(BIND, (P, f, &asmb, pos, need)) // extend obj by set a binding
194  // http://piumarta.com/software/cola/colas-whitepaper.pdf
195  CASE_OP(MESSAGE, (P, f, &asmb, pos, need))// call a method of an object
196  CASE_OP(JMP, (P, f, &asmb, pos, jmps, offs, &jmpc)) // PC += sBx
197  CASE_OP(TEST, (P, f, &asmb, pos)) // if not (R(A) <=> C) then PC++
198  CASE_OP(NOT, (P, f, &asmb, pos)) // a = not b
199  CASE_OP(CMP, (P, f, &asmb, pos))
200  CASE_OP(TESTJMP, (P, f, &asmb, pos, jmps, offs, &jmpc))
201  CASE_OP(NOTJMP, (P, f, &asmb, pos, jmps, offs, &jmpc))
202  CASE_OP(NAMED, (P, f, &asmb, pos, need)) // assign named args before a CALL
203  CASE_OP(CALL, (P, f, &asmb, pos, need)) // call a function. R(A),...:= R(A)(R(A+1),...,R(A+B-1))
204  CASE_OP(CALLSET, (P, f, &asmb, pos, need))//? set return register to write to
205  CASE_OP(RETURN, (P, f, &asmb, pos)) // return R(A), ... ,R(A+B-2)
206  CASE_OP(PROTO, (P, f, &asmb, &pos, lregs, need, regs))// define function prototype
207  CASE_OP(CLASS, (P, f, &asmb, pos, need)) // find class for register value
208  }

230 PN potion_vm(Potion *P, PN proto, PN self, PN vargs, PN_SIZE upc, PN *upargs) {
231  vPN(Proto) f = (struct PNProto *)proto;
232 
233  // these variables persist as we jump around
234  PN stack[STACK_MAX];
235  PN val = PN_NIL;
236 
237  // these variables change from proto to proto
238  // current = upvals | locals | self | reg
239  PN_SIZE pos = 0;
240  long argx = 0;
241  PN *args = NULL, *upvals, *locals, *reg;
242  PN *current = stack;
243 
244  if (vargs != PN_NIL) args = PN_GET_TUPLE(vargs)->set;
245 reentry:
246  if (current - stack >= STACK_MAX) {
247  fprintf(stderr, "all registers used up!");
248  exit(1);
249  }
250 
251  upvals = current;
252  locals = upvals + f->upvalsize;
253  reg = locals + f->localsize + 1;
254 
255  if (pos == 0) {
256  reg[-1] = reg[0] = self;
257  if (f->localsize)
258  memset((void*)locals, 0, sizeof(PN) * f->localsize);
259  if (upc > 0 && upargs != NULL) {
260  PN_SIZE i;
261  for (i = 0; i < upc; i++) {
262  upvals[i] = upargs[i];
263  }
264  }
265 
266  if (args != NULL) {
267  argx = 0;
268  PN_TUPLE_EACH(f->sig, i, v, {
269  if (PN_IS_STR(v)) {
270  PN_SIZE num = PN_GET(f->locals, v);
271  locals[num] = args[argx++];
272  }
273  });
274  }
275  }
276 
277 #ifdef DEBUG
278  if (P->flags & DEBUG_TRACE)
279  fprintf(stderr, "-- run-time --\n");
280 #endif
281 
282  while (pos < PN_OP_LEN(f->asmb)) {
283  PN_OP op = PN_OP_AT(f->asmb, pos);
284 #ifdef DEBUG
285  if (P->flags & DEBUG_TRACE) {
286  fprintf(stderr, "[%2d] %-8s %d", pos+1, potion_ops[op.code].name, op.a);
287  if (potion_ops[op.code].args > 1)
288  fprintf(stderr, " %d", op.b);
289  }
290 #endif
291  // TODO: cgoto (does not check boundaries, est. ~10-20% faster)
292  switch (op.code) {
293  case OP_MOVE:
294  reg[op.a] = reg[op.b];
295  break;
296  case OP_LOADK:
297  reg[op.a] = PN_TUPLE_AT(f->values, op.b);
298  break;
299  case OP_LOADPN:
300  reg[op.a] = (PN)op.b;
301  break;
302  case OP_SELF:
303  reg[op.a] = reg[-1];
304  break;
305  case OP_GETLOCAL:
306  if (PN_IS_REF(locals[op.b]))
307  reg[op.a] = PN_DEREF(locals[op.b]);
308  else
309  reg[op.a] = locals[op.b];
310  break;
311  case OP_SETLOCAL:
312  if (PN_IS_REF(locals[op.b])) {
313  PN_DEREF(locals[op.b]) = reg[op.a];
314  PN_TOUCH(locals[op.b]);
315  } else
316  locals[op.b] = reg[op.a];
317  break;
318  case OP_GETUPVAL:
319  reg[op.a] = PN_DEREF(upvals[op.b]);
320  break;
321  case OP_SETUPVAL:
322  PN_DEREF(upvals[op.b]) = reg[op.a];
323  PN_TOUCH(upvals[op.b]);
324  break;
325  case OP_GLOBAL:
326  potion_define_global(P, reg[op.a], reg[op.b]);
327  reg[op.a] = reg[op.b];
328  break;
329  case OP_NEWTUPLE:
330  reg[op.a] = PN_TUP0();
331  break;
332  case OP_SETTUPLE:
333  reg[op.a] = PN_PUSH(reg[op.a], reg[op.b]);
334  break;
335  case OP_SETTABLE:
336  potion_table_set(P, reg[op.a], reg[op.a + 1], reg[op.b]);
337  break;
338  case OP_NEWLICK: {
339  PN attr = op.b > op.a ? reg[op.a + 1] : PN_NIL;
340  PN inner = op.b > op.a + 1 ? reg[op.b] : PN_NIL;
341  reg[op.a] = potion_lick(P, reg[op.a], attr, inner);
342  }
343  break;
344  case OP_GETPATH:
345  reg[op.a] = potion_obj_get(P, PN_NIL, reg[op.a], reg[op.b]);
346  break;
347  case OP_SETPATH:
348  potion_obj_set(P, PN_NIL, reg[op.a], reg[op.a + 1], reg[op.b]);
349  break;
350  case OP_ADD:
351  PN_VM_MATH(add, +);
352  break;
353  case OP_SUB:
354  PN_VM_MATH(sub, -);
355  break;
356  case OP_MULT:
357  PN_VM_MATH(mult, *);
358  break;
359  case OP_DIV:
360  PN_VM_MATH(div, /);
361  break;
362  case OP_REM:
363  PN_VM_MATH(rem, %);
364  break;
365  case OP_POW:
366  reg[op.a] = PN_NUM((int)pow((double)PN_INT(reg[op.a]),
367  (double)PN_INT(reg[op.b])));
368  break;
369  case OP_NOT:
370  reg[op.a] = PN_BOOL(!PN_TEST(reg[op.a]));
371  break;
372  case OP_CMP:
373  reg[op.a] = PN_NUM(PN_INT(reg[op.b]) - PN_INT(reg[op.a]));
374  break;
375  case OP_NEQ:
376  reg[op.a] = PN_BOOL(reg[op.a] != reg[op.b]);
377  break;
378  case OP_EQ:
379  reg[op.a] = PN_BOOL(reg[op.a] == reg[op.b]);
380  break;
381  case OP_LT:
382  reg[op.a] = PN_BOOL((long)(reg[op.a]) < (long)(reg[op.b]));
383  break;
384  case OP_LTE:
385  reg[op.a] = PN_BOOL((long)(reg[op.a]) <= (long)(reg[op.b]));
386  break;
387  case OP_GT:
388  reg[op.a] = PN_BOOL((long)(reg[op.a]) > (long)(reg[op.b]));
389  break;
390  case OP_GTE:
391  reg[op.a] = PN_BOOL((long)(reg[op.a]) >= (long)(reg[op.b]));
392  break;
393  case OP_BITN:
394  reg[op.a] = PN_IS_NUM(reg[op.b]) ? PN_NUM(~PN_INT(reg[op.b])) : potion_obj_bitn(P, reg[op.b]);
395  break;
396  case OP_BITL:
397  PN_VM_MATH(bitl, <<);
398  break;
399  case OP_BITR:
400  PN_VM_MATH(bitr, >>);
401  break;
402  case OP_DEF:
403  reg[op.a] = potion_def_method(P, PN_NIL, reg[op.a], reg[op.a + 1], reg[op.b]);
404  break;
405  case OP_BIND:
406  reg[op.a] = potion_bind(P, reg[op.b], reg[op.a]);
407  break;
408  case OP_MESSAGE:
409  reg[op.a] = potion_message(P, reg[op.b], reg[op.a]);
410  break;
411  case OP_JMP:
412  pos += op.a;
413  break;
414  case OP_TEST:
415  reg[op.a] = PN_BOOL(PN_TEST(reg[op.a]));
416  break;
417  case OP_TESTJMP:
418  if (PN_TEST(reg[op.a])) pos += op.b;
419  break;
420  case OP_NOTJMP:
421  if (!PN_TEST(reg[op.a])) pos += op.b;
422  break;
423  case OP_NAMED: {
424  int x = potion_sig_find(P, reg[op.a], reg[op.b - 1]);
425  if (x >= 0) reg[op.a + x + 2] = reg[op.b];
426  }
427  break;
428  case OP_CALL:
429  switch (PN_TYPE(reg[op.a])) {
430  case PN_TVTABLE:
431  reg[op.a + 1] = potion_object_new(P, PN_NIL, reg[op.a]);
432  reg[op.a] = ((struct PNVtable *)reg[op.a])->ctor;
433  case PN_TCLOSURE:
434  if (PN_CLOSURE(reg[op.a])->method != (PN_F)potion_vm_proto) {
435  reg[op.a] = potion_call(P, reg[op.a], op.b - op.a, reg + op.a + 1);
436  } else if (((reg - stack) + PN_INT(f->stack) + f->upvalsize + f->localsize + 8) >= STACK_MAX) {
437  int i;
438  PN argt = potion_tuple_with_size(P, (op.b - op.a) - 1);
439  for (i = 2; i < op.b - op.a; i++)
440  PN_TUPLE_AT(argt, i - 2) = reg[op.a + i];
441  reg[op.a] = potion_vm(P, PN_CLOSURE(reg[op.a])->data[0], reg[op.a + 1], argt,
442  PN_CLOSURE(reg[op.a])->extra - 1, &PN_CLOSURE(reg[op.a])->data[1]);
443  } else {
444  self = reg[op.a + 1];
445  args = &reg[op.a + 2];
446  upc = PN_CLOSURE(reg[op.a])->extra - 1;
447  upargs = &PN_CLOSURE(reg[op.a])->data[1];
448  current = reg + PN_INT(f->stack) + 2;
449  current[-2] = (PN)f;
450  current[-1] = (PN)pos;
451 
452  f = PN_PROTO(PN_CLOSURE(reg[op.a])->data[0]);
453  pos = 0;
454  goto reentry;
455  }
456  break;
457 
458  default: {
459  reg[op.a + 1] = reg[op.a];
460  reg[op.a] = potion_obj_get_call(P, reg[op.a]);
461  if (PN_IS_CLOSURE(reg[op.a]))
462  reg[op.a] = potion_call(P, reg[op.a], op.b - op.a, &reg[op.a + 1]);
463  }
464  break;
465  }
466  break;
467  case OP_CALLSET:
468  reg[op.a] = potion_obj_get_callset(P, reg[op.b]);
469  break;
470  case OP_RETURN:
471  if (current != stack) {
472  val = reg[op.a];
473 
474  f = PN_PROTO(current[-2]);
475  pos = (PN_SIZE)current[-1];
476  op = PN_OP_AT(f->asmb, pos);
477 
478  reg = current - (PN_INT(f->stack) + 2);
479  current = reg - (f->localsize + f->upvalsize + 1);
480  reg[op.a] = val;
481  pos++;
482  goto reentry;
483  } else {
484  reg[0] = reg[op.a];
485  goto done;
486  }
487  break;
488  case OP_PROTO: {
489  vPN(Closure) cl;
490  unsigned areg = op.a;
491  proto = PN_TUPLE_AT(f->protos, op.b);
492  cl = (struct PNClosure *)potion_closure_new(P, (PN_F)potion_vm_proto,
493  PN_PROTO(proto)->sig, PN_TUPLE_LEN(PN_PROTO(proto)->upvals) + 1);
494  cl->data[0] = proto;
495  PN_TUPLE_COUNT(PN_PROTO(proto)->upvals, i, {
496  pos++;
497  op = PN_OP_AT(f->asmb, pos);
498 
499  if (op.code == OP_GETUPVAL) {
500  cl->data[i+1] = upvals[op.b];
501  } else if (op.code == OP_GETLOCAL) {
502  cl->data[i+1] = locals[op.b] = (PN)potion_ref(P, locals[op.b]);
503  } else {
504  fprintf(stderr, "** missing an upval to proto %p\n", (void *)proto);
505  }
506  });
507  reg[areg] = (PN)cl;
508  }
509  break;
510  case OP_CLASS:
511  reg[op.a] = potion_vm_class(P, reg[op.b], reg[op.a]);
512  break;
513  }
