87 PN potion_callcc(Potion *P, PN cl, PN self) {
88  struct PNCont *cc;
89  PN_SIZE n;
90  PN *start, *sp1 = P->mem->cstack, *sp2, *sp3;
91  POTION_ESP(&sp2);
92  POTION_EBP(&sp3);
93 #if POTION_STACK_DIR > 0
94  n = sp2 - sp1;
95  start = sp1;
96 #else
97  n = sp1 - sp2 + 1;
98  start = sp2;
99 #endif
100 
101  cc = PN_ALLOC_N(PN_TCONT, struct PNCont, sizeof(PN) * (n + 3 + PN_SAVED_REGS));
102  cc->len = n + 3;
103  cc->stack[0] = (PN)sp1;
104  cc->stack[1] = (PN)sp2;
105  cc->stack[2] = (PN)sp3;
106  cc->stack[3] = PN_NIL;
107 #if POTION_X86 == POTION_JIT_TARGET
108 #if __WORDSIZE == 64
109  __asm__ ("mov %%rbx, 0x20(%0);"
110  "mov %%r12, 0x28(%0);"
111  "mov %%r13, 0x30(%0);"
112  "mov %%r14, 0x38(%0);"
113  "mov %%r15, 0x40(%0);"::"r"(cc->stack));
114 #else
115  __asm__ ("mov %%esi, 0x10(%0);"
116  "mov %%edi, 0x14(%0);"
117  "mov %%ebx, 0x18(%0)"::"r"(cc->stack));
118 #endif
119 #endif
120 
121 // avoid wrong asan stack underflow, caught in memcpy
122 #if defined(__clang__) && defined(__SANITIZE_ADDRESS__)
123  {
124  PN *s = start + 1;
125  PN *d = cc->stack + 4 + PN_SAVED_REGS;
126  for (int i=0; i < n - 1; i++) {
127  *d++ = *s++;
128  }
129  }
130 #else
131  PN_MEMCPY_N((char *)(cc->stack + 4 + PN_SAVED_REGS), start + 1, PN, n - 1);
132 #endif
133  return (PN)cc;
134 }
