A = (i, j):
  ij = i + j
  1.0 / (ij * (ij + 1) / 2 + i + 1).

Av = (n, x, y):
  i = 0
  while (i < n):
    a = 0, j = 0
    while (j < n):
      a += A (i, j) * x (j)
      j++.
    y (i) = a
    i++.
  .

Atv = (n, x, y):

    y (i) = a
    i++.
  .

AtAv = (n, x, y, t):
