70 static void pcg32_srandom_r(pcg32_random_t* rng, uint64_t initstate, uint64_t initseq)
71 {
72  rng->state = 0U;
73  rng->inc = (initseq << 1u) | 1u;
74  pcg32_random_r(rng);
75  rng->state += initstate;
76  pcg32_random_r(rng);
77 }

85  * pcg32_random_r(rng)
86  * Generate a uniformly distributed 32-bit random number */
87 
88 static uint32_t pcg32_random_r(pcg32_random_t* rng)
89 {
90  uint64_t oldstate = rng->state;
91  rng->state = oldstate * 6364136223846793005ULL + rng->inc;
92  uint32_t xorshifted = ((oldstate >> 18u) ^ oldstate) >> 27u;
93  uint32_t rot = oldstate >> 59u;
94  return (xorshifted >> rot) | (xorshifted << ((-rot) & 31));
95 }

103 /* pcg32_boundedrand(bound):
104  * pcg32_boundedrand_r(rng, bound):
105  * Generate a uniformly distributed number, r, where 0 <= r < bound */
106 
107 static uint32_t pcg32_boundedrand_r(pcg32_random_t* rng, uint32_t bound)
108 {
109  /* To avoid bias, we need to make the range of the RNG a multiple of
110  * bound, which we do by dropping output less than a threshold.
111  * A naive scheme to calculate the threshold would be to do
112  *
113  * uint32_t threshold = 0x100000000ull % bound;
114  *
115  * but 64-bit div/mod is slower than 32-bit div/mod (especially on
116  * 32-bit platforms). In essence, we do
117  *
118  * uint32_t threshold = (0x100000000ull-bound) % bound;
119  *
120  * because this version will calculate the same modulus, but the LHS
121  * value is less than 2^32. */
122 
123  uint32_t threshold = -bound % bound;
124 
125  /* Uniformity guarantees that this loop will terminate. In practice, it
126  * should usually terminate quickly; on average (assuming all bounds are
127  * equally likely), 82.25% of the time, we can expect it to require just
128  * one iteration. In the worst case, someone passes a bound of 2^31 + 1
129  * (i.e., 2147483649), which invalidates almost 50% of the range. In
130  * practice, bounds are typically small and only a tiny amount of the range
131  * is eliminated. */
132  for (;;) {
133  uint32_t r = pcg32_random_r(rng);
134  if (r >= threshold)
135  return r % bound;
136  }
137 }
