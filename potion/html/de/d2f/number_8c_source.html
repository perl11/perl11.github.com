299 static PN potion_int_to(Potion *P, PN cl, PN self, PN end, PN block) {
300  long i, s = 1, j = PN_INT(self), k = PN_INT(end);
301  PN_CHECK_INT(self);
302  PN_CHECK_INT(end);

306  for (i = j; i != k + s; i += s)

308  return PN_NUM(labs(i - j));
