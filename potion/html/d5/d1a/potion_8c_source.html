71 static PN potion_cmd_exec(Potion *P, PN buf, char *filename, char *compile, char *addcode) {
72  exec_mode_t exec = (exec_mode_t)(P->flags & ((1<<EXEC_BITS)-1));
73  int fd = -1;
74  PN code = PN_NIL;
75 
76  if (!buf && filename) {
77  struct stat stats;
78  if (stat(filename, &stats) == -1) {
79  fprintf(stderr, "** %s does not exist.", filename);
80  goto done;
81  }
82  fd = open(filename, O_RDONLY | O_BINARY);
83  if (fd == -1) {
84  fprintf(stderr, "** could not open %s. check permissions.", filename);
85  goto done;
86  }
87  long size = stats.st_size;
88  char *bufptr;
89  if (addcode) {
90  int len = strlen(addcode);
91  size += len;
92  buf = potion_bytes(P, size);
93  bufptr = PN_STR_PTR(buf);
94  PN_MEMCPY_N(bufptr, addcode, char, len);
95  bufptr += len;
96  } else {
97  buf = potion_bytes(P, size);
98  bufptr = PN_STR_PTR(buf);
99  }
100  // TODO: mmap instead of read all
101  if (read(fd, bufptr, stats.st_size) == stats.st_size) {
102  PN_STR_PTR(buf)[size] = '\0';
103  } else {
104  fprintf(stderr, "** could not read entire file.");
105  goto done;
106  }
107  }
108  else if (!filename) filename = "-e";
109 
110  code = potion_source_load(P, PN_NIL, buf);
111  if (PN_IS_PROTO(code)) {
112  DBG_v("\n-- loaded --\n");
113  } else {
114  code = potion_parse(P, buf, filename);
115  if (!code || PN_TYPE(code) == PN_TERROR) {
116  potion_p(P, code);
117  return code;
118  }
119  DBG_v("\n-- parsed --\n");
120  DBG_Pv(code);
121  code = potion_send(code, PN_compile, potion_str(P, filename),
122  compile ? potion_str(P, compile): PN_NIL);
123  DBG_v("\n-- compiled --\n");
124  }
125  DBG_Pv(code);
126  if (exec == EXEC_VM || exec == EXEC_DEBUG) {
127  code = potion_vm(P, code, P->lobby, PN_NIL, 0, NULL);
128  DBG_v("\n-- vm returned %p (fixed=%ld, actual=%ld, reserved=%ld, time=%0.6gms %dx/%dm/%di) --\n",
129  (void *)code,
130  PN_INT(potion_gc_fixed(P, 0, 0)), PN_INT(potion_gc_actual(P, 0, 0)),
131  PN_INT(potion_gc_reserved(P, 0, 0)), P->mem->time * 1000, P->mem->pass,
132  P->mem->majors, P->mem->minors);
133  DBG_Pvi(code);
134  } else if (exec == EXEC_JIT) {
135 #ifdef POTION_JIT_TARGET
136  PN val;
137  PN cl = potion_closure_new(P, (PN_F)potion_jit_proto(P, code), PN_NIL, 1);
138  PN_CLOSURE(cl)->data[0] = code;
139  val = PN_PROTO(code)->jit(P, cl, P->lobby);
140  DBG_v("\n-- jit returned %p (fixed=%ld, actual=%ld, reserved=%ld, time=%0.6gms %dx/%dm/%di) --\n", PN_PROTO(code)->jit,
141  PN_INT(potion_gc_fixed(P, 0, 0)), PN_INT(potion_gc_actual(P, 0, 0)),
142  PN_INT(potion_gc_reserved(P, 0, 0)), P->mem->time * 1000, P->mem->pass,
143  P->mem->majors, P->mem->minors);
144  DBG_Pvi(val);
145 #else
146  fprintf(stderr, "** potion built without JIT support\n");
147 #endif
148  }
149  else if (exec == EXEC_CHECK) {
150  DBG_v("\n-- check --\n");
151  }
152 
153  if (!code || PN_TYPE(code) == PN_TERROR)
154  goto done;
155 
156  if (exec >= MAX_EXEC)
157  potion_fatal("fatal: stack corruption (exec > MAX_EXEC)\n");
158 
159  if (exec == EXEC_COMPILE) { // needs an inputfile. TODO: -e"" -ofile
160  char outpath[255];
161  FILE *pnb;
162  PN opts;
163  char *c_opts = NULL;
164  PN_SIZE written = 0;
165  if (compile) { // --compile=c[,OPTS]
166  if ((c_opts = strchr(compile,','))) {
167  c_opts[0] = '\0';
168  c_opts++;
169  //TODO check -o for outpath
170  }
171  }
172  if (!compile || !strcmp(compile, "bc"))
173  sprintf(outpath, "%sb", filename); // .pnb
174  else if (!strcmp(compile, "c"))
175  sprintf(outpath, "%s.c", filename); // .pn.c
176  else if (!strcmp(compile, "exe"))
177  sprintf(outpath, "%s.out", filename); // TODO: strip ext
178  opts = c_opts
179  ? pn_printf(P, potion_bytes(P,0), "%s,-o%s", c_opts, outpath)
180  : potion_strcat(P, "-o", outpath);
181 
182  pnb = fopen(outpath, "wb");
183  if (!pnb) {
184  fprintf(stderr, "** could not open %s for writing. check permissions.\n", outpath);
185  goto done;
186  }
187  if (!compile)
188  code = potion_source_dumpbc(P, 0, code, PN_NIL);
189  else
190  code = potion_source_dump(P, 0, code,
191  potion_str(P, compile),
192  opts ? opts : PN_NIL);
193  if (code &&
194  (written = fwrite(PN_STR_PTR(code), 1, PN_STR_LEN(code), pnb) == PN_STR_LEN(code))) {
195  printf("** compiled code saved to %s\n", outpath);
196  fclose(pnb);
197 
198  if (!compile || !strcmp(compile, "bc"))
199  printf("** run it with: potion %s\n", outpath);
200  // TODO: let the compilers write its own hints (,-ooutfile)
201  else if (!strcmp(compile, "c"))
202  printf("** compile it with: %s %s %s\n", POTION_CC, POTION_CFLAGS, outpath);
203  else if (!strcmp(compile, "exe"))
204  printf("** run it with: ./%s\n", outpath);
205  } else {
206  fprintf(stderr, "** could not write all %s compiled code (%u/%u) to %s\n",
207  compile?compile:"bytecode", written, code?PN_STR_LEN(code):0, outpath);
208  }
209  }
210 
211 #if defined(DEBUG)
212  if (P->flags & DEBUG_GC) { // GC sanity check
213  fprintf(stderr, "\n-- gc check --\n");
214  void *scanptr = (void *)((char *)P->mem->old_lo + (sizeof(PN) * 2));
215  while ((PN)scanptr < (PN)P->mem->old_cur) {
216  fprintf(stderr, "%p.vt = %x (%u)\n",
217  scanptr, ((struct PNObject *)scanptr)->vt,
218  potion_type_size(P, scanptr));
219  if (((struct PNFwd *)scanptr)->fwd != POTION_FWD
220  && ((struct PNFwd *)scanptr)->fwd != POTION_COPIED) {
221  PNType vt = ((struct PNObject *)scanptr)->vt;
222  if ((signed int)vt < 0 || vt > PN_TUSER) {
223  fprintf(stderr, "** wrong type for allocated object: %p.vt = %x\n",
224  scanptr, vt);
225  break;
226  }
227  }
228  scanptr = (void *)((char *)scanptr + potion_type_size(P, scanptr));
229  if ((PN)scanptr > (PN)P->mem->old_cur) {
230  fprintf(stderr, "** allocated object goes beyond GC pointer\n");
231  break;
232  }
233  }
234  }
235 #endif
236 
237 done:
238  if (fd != -1)
239  close(fd);
240 
241  return code;
242 }

260 int main(int argc, char *argv[]) {
261  POTION_INIT_STACK(sp);
262  int i;
263  exec_mode_t exec = POTION_JIT ? EXEC_JIT : EXEC_VM;
264  Potion *P = potion_create(sp);
265  PN buf = PN_NIL;
266  char *compile = NULL;
267  char *fn = NULL;
268  char *addmodules = NULL;
269 
270  for (i = 1; i < argc; i++) {
271  if (!strcmp(argv[i], "--")) { i++; break; }
272  if (!strcmp(argv[i], "-I") || !strcmp(argv[i], "--inspect")) {
273  P->flags |= DEBUG_INSPECT; continue; }
274  if (!strcmp(argv[i], "-L")) {
275  char *extra_path = &argv[i][2]; // todo: flexible
276  if (*extra_path)
277  potion_loader_add(P, potion_str(P, extra_path));
278  else {
279  if (i == argc) {
280  fprintf(stderr, "-L missing directory\n");
281  goto END;
282  }
283  potion_loader_add(P, potion_str(P, argv[i+1]));
284  i++;
285  }
286  continue;
287  }
288  if (!strcmp(argv[i], "-V") || !strcmp(argv[i], "--verbose")) {
289  P->flags |= DEBUG_VERBOSE; continue; }
290  if (!strcmp(argv[i], "-v") || !strcmp(argv[i], "--version")) {
291  potion_cmd_version(P); goto END; }
292  if (!strcmp(argv[i], "-h") || !strcmp(argv[i], "--help")) {
293  potion_cmd_usage(P); goto END; }
294  if (!strcmp(argv[i], "-s") || !strcmp(argv[i], "--stats")) {
295  potion_cmd_stats(P); goto END; }
296  if (!strcmp(argv[i], "--check")) { exec = EXEC_CHECK; continue; }
297  if (!strncmp(argv[i], "--compile=", 10)) {
298  exec = EXEC_COMPILE; compile = &argv[i][10]; continue; }
299  if (!strcmp(argv[i], "--compile") || !strcmp(argv[i], "-c")) {
300  exec = EXEC_COMPILE; compile = NULL; continue; }
301  if (!strcmp(argv[i], "-B") || !strcmp(argv[i], "--bytecode")) {
302  exec = EXEC_VM; continue; }
303  if (!strcmp(argv[i], "-X") || !strcmp(argv[i], "--x86")) {
304  exec = EXEC_JIT; continue; }
305  if (!strcmp(argv[i], "--debug") || !strcmp(argv[i], "-d")) {
306  exec = EXEC_DEBUG; continue; }
307 #ifdef DEBUG
308  if (argv[i][0] == '-' && argv[i][1] == 'D') {
309  if (strchr(&argv[i][2], '?')) {
310  printf("-D debugging flags:\n");
311  printf(" i inspect\n");
312  printf(" v verbose\n");
313  printf(" t trace\n");
314  printf(" p parse\n");
315  printf(" P parse verbose\n");
316  printf(" c compile\n");
317  printf(" J disassemble Jit code\n");
318  printf(" G GC (use w/ or wo/ -Dv\n");
319  goto END;
320  }
321  if (strchr(&argv[i][2], 'i')) P->flags |= DEBUG_INSPECT;
322  if (strchr(&argv[i][2], 'v')) P->flags |= DEBUG_VERBOSE;
323  if (strchr(&argv[i][2], 't')) { P->flags |= DEBUG_TRACE;
324  exec = exec==EXEC_JIT ? EXEC_VM : exec; }
325  if (strchr(&argv[i][2], 'p')) P->flags |= DEBUG_PARSE;
326  if (strchr(&argv[i][2], 'P')) P->flags |= DEBUG_PARSE_VERBOSE;
327  if (strchr(&argv[i][2], 'c')) P->flags |= DEBUG_COMPILE;
328  if (strchr(&argv[i][2], 'J')) P->flags |= DEBUG_JIT;
329  if (strchr(&argv[i][2], 'G')) P->flags |= DEBUG_GC;
330  continue;
331  }
332 #endif
333  if (argv[i][0] == '-') {
334  if (argv[i][1] == 'e') {
335  char *arg;
336  if (strlen(argv[i]) == 2) {
337  arg = argv[i+1];
338  } else if (i <= argc) {
339  arg = argv[i]+2;
340  } else { // or go into interactive mode?
341  potion_fatal("Missing argument for -e");
342  goto END;
343  }
344  if (addmodules)
345  buf = potion_strcat(P, addmodules, arg);
346  else
347  buf = potion_str(P, arg);
348  continue;
349  } else {
350  fprintf(stderr, "** Unrecognized option: %s\n", argv[i]);
351  }
352  }
353  else {
354  break;
355  }
356  }
357  P->flags = P->flags + exec;
358 
359  if (buf || i < argc) {
360  PN args = PN_TUP0();
361  if (buf == PN_NIL) { fn = argv[i++]; PN_PUSH(args, PN_STR(fn)); }
362  else { PN_PUSH(args, PN_STR("-e")); }
363  for (; i < argc; i++) PN_PUSH(args, PN_STR(argv[i]));
364  potion_define_global(P, PN_STR("argv"), args);
365  if (buf != PN_NIL) {
366  potion_cmd_exec(P, buf, NULL, compile, PN_NIL);
367  } else {
368  potion_cmd_exec(P, buf, fn, compile, addmodules);
369  //potion_cmd_compile(P, fn, compile);
370  }
371  } else {
372  if (!exec || P->flags & DEBUG_INSPECT) potion_fatal("no filename given");
373  PN args = PN_TUP0();
374  PN_PUSH(args, PN_STR(""));
375  potion_define_global(P, PN_STR("argv"), args);
376  potion_eval(P, potion_byte_str(P,
377  "load 'readline'\n" \
378  "loop:\n" \
379  " code = readline('>> ')\n" \
380  " if (not code): \"\\n\" print, break.\n" \
381  " if (code != ''):\n" \
382  " obj = code eval\n" \
383  " if (obj kind == Error):\n" \
384  " obj string print." \
385  " else: ('=> ', obj, \"\\n\") join print.\n" \
386  " .\n"
387  "."));
388  }
389  END:
390 #if !defined(POTION_JIT_TARGET) || defined(DEBUG)
391  if (P != NULL)
392  potion_destroy(P);
393 #endif
394  return 0;
395 }
